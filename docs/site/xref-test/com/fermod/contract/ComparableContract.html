1   package com.fermod.contract;

3   import static org.junit.jupiter.api.Assertions.assertEquals;
4   import static org.junit.jupiter.api.Assertions.assertTrue;
5   
6   import org.junit.jupiter.api.DisplayName;
7   import org.junit.jupiter.api.Tag;
8   import org.junit.jupiter.api.Test;
9   
10  /**
11   * This interface defines tests to verify that the objects of each class that 
12   * implements the {@link Comparable} interface, impose their total ordering with
13   * the expected behavior. This ordering is referred to as the classes 
14   * <i>natural ordering</i>, and the classes {@code compareTo} method is referred
15   * to as its <i>natural comparison method</i>.
16   * <p>
17   * This class extends from {@link Testable}, and in order to function correctly 
18   * all the required methods must be implemented.
19   *
20   * @param <T> the type of objects that this object may be compared to
21   * 
22   * @see Testable
23   */
24  @DisplayName("Comparable Test")
25  @Tag("comparable")
26  public interface ComparableContract<T extends Comparable<T>> extends Testable<T> {
27  	
28  	/**
29  	 * Creates and returns a smaller value to use in the comparable tests.<br>
30  	 * This method, for its correct execution, should return a value smaller 
31  	 * than the returned by {@link Testable#createValue}.
32  	 * 
33  	 * @return the created value
34  	 */
35  	T createSmallerValue();
36  
37  	/**
38  	 * Compares the created value to itself to test that returns a zero value.
39  	 */
40  	@DisplayName("Compare Itself - Return Zero")
41  	@Test
42  	default void comparedToItself() {
43  		T value = createValue();
44  		assertEquals(0, value.compareTo(value));
45  	}
46  
47  	/**
48  	 * Compares the created value to a smaller value to test that returns a 
49  	 * positive value.
50  	 */
51  	@DisplayName("Compare To Smaller - Return Positive Number")
52  	@Test
53  	default void compareToSmallerValue() {
54  		T value = createValue();
55  		T smallerValue = createSmallerValue();
56  		assertTrue(value.compareTo(smallerValue) > 0);
57  	}
58  
59  	/**
60  	 * Compares the created value to a larger value to test that returns a 
61  	 * negative value.
62  	 */
63  	@DisplayName("Compare To Larger - Return Negative Number")
64  	@Test
65  	default void compareToLargerValue() {
66  		T value = createValue();
67  		T smallerValue = createSmallerValue();
68  		assertTrue(smallerValue.compareTo(value) < 0);
69  	}
70  
71  }
